Code Sample: Computing YoY Growth

def yoy_growth(current, prior):
    """Return YoY growth (%) with safe divide."""
    if prior == 0:
        return float('inf')
    return (current / prior - 1.0) * 100.0

print(f"Revenue YoY: {yoy_growth(4320, 3855):.2f}%")
